agot_filler_north.0001 = { # Encounter Wildling
	type = character_event
	title = agot_filler_north.0001.t
	desc = agot_filler_north.0001.desc

	theme = hunt_activity

	override_background = {
		reference = wilderness_forest_pine
	}

	left_portrait = {
		character = root
		animation = disgust
	}

	right_portrait = {
		character = scope:agot_filler_wildling_male
		animation = personality_callous
	}

	trigger = {
		is_available = yes
		is_at_war = no
		effective_age >= 8
		any_held_county = {
			title_province = { geographical_region = world_westeros_the_north }
		}
		NOT = {
			any_held_county = {
				title_province = { geographical_region = world_westeros_the_neck }
			}
			has_character_flag = agot_filler_wildling_encounter
		}
	}

	immediate = {
		add_character_flag = {
			flag = agot_filler_wildling_encounter
			years = 20
		}
		create_character = {
			location = root.capital_province
			template = agot_wildling_character
			save_scope_as = agot_filler_wildling_male
			gender = root
		}
	}

	# With Wildlings, its kill or be killed...
	option = {
		name = agot_filler_north.0001.a
		custom_tooltip = agot_filler_north.0001.a.tt
		trigger = {
			effective_age >=16
			NOT = { has_trait = compassionate }
		}
		configure_start_single_combat_effect = {
			SC_INITIATOR = root
			SC_ATTACKER = root
			SC_DEFENDER = scope:agot_filler_wildling_male
			FATALITY = always
			FIXED = no
			LOCALE = terrain_scope
			OUTPUT_EVENT = single_combat.1006
			INVALIDATION_EVENT = single_combat.1006
		}
		#remove_variable = duel_encounter
	}

	# The Wildlings are just like us!
	option = {
		name = agot_filler_north.0001.b
		custom_tooltip = agot_filler_north.0001.b.tt
		trigger = {
			is_ai = no
			has_trait = compassionate
			NOR = {
				has_trait = callous
				has_trait = paranoid
			}
		}
		trait = compassionate
		trigger_event = {
			id = agot_filler_north.0002
		}
	}

		# Try to hide
	option = {
		name = agot_filler_north.0001.c
		custom_tooltip = agot_filler_north.0001.c.tt
		add_prestige = miniscule_prestige_loss
		if = {
			limit = {
				number_of_personality_traits < personality_trait_limit
				NOR = {
					has_trait = craven
					has_trait = brave
				}
			}
			random = {
				chance = 5
				add_trait = craven
			}
		}
		if = {
			limit = {
				number_of_personality_traits < personality_trait_limit
				NOR = {
					has_trait = paranoid
					has_trait = trusting
				}
			}
			random = {
				chance = 5
				add_trait = paranoid
			}
		}

	}
}
