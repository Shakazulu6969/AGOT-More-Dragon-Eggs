on_travel_move_dragon_pit_location = {
    effect = {
        every_courtier_or_guest = { #take everyone you have with you
			limit = {
				has_trait = dragon
                has_character_flag = in_dragonpit
			}
			save_temporary_scope_as = temp_dragon
			agot_send_to_dragonpit_no_event = {
				DRAGON = scope:temp_dragon
				DRAGONPIT_COUNTY = root.domicile.domicile_location.county
			}
		}
    }
}
