agot_dragon_tree_creation_effect = {
	$DRAGON$ = { save_scope_as = dragon }
	if = {
		limit = {
			exists = scope:dragon.mother
			scope:dragon.mother = { has_trait = dragon }
		}
		agot_add_to_dragon_tree = { PARENT = scope:dragon.mother CHILD = scope:dragon }
	}
	else_if = {
		limit = {
			exists = character:dragon_vermax
			exists = scope:dragon.father
			scope:dragon.father = character:dragon_vermax
		}
		agot_add_to_dragon_tree = { PARENT = character:dragon_vermax CHILD = scope:dragon }
	}
    else_if = {
		limit = {
			exists = character:dragon_cannibal
			exists = scope:dragon.father
			scope:dragon.father = character:dragon_cannibal
		}
		agot_add_to_dragon_tree = { PARENT = character:dragon_cannibal CHILD = scope:dragon }
	}
    else_if = {
		limit = {
			exists = character:dragon_sheepstealer
			exists = scope:dragon.father
			scope:dragon.father = character:dragon_sheepstealer
		}
		agot_add_to_dragon_tree = { PARENT = character:dragon_sheepstealer CHILD = scope:dragon }
	}
	else_if = {
		limit = {
			exists = character:dragon_vassarion
			exists = scope:dragon.father
			scope:dragon.father = character:dragon_vassarion
		}
		agot_add_to_dragon_tree = { PARENT = character:dragon_vassarion CHILD = scope:dragon }
	}
}
